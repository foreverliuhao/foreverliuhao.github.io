package com.hello.leetcode;

import java.util.ArrayList;
import java.util.List;
import java.util.Scanner;

/**

  • 问题描述：在一条直线上，有n个房屋，
  • 每个房屋中数量不等的财宝，有一个盗贼希望从房屋中盗取财宝，由于房屋中有报警器，如果同时从相邻的两个房屋中盗取财宝就会触发报警器。问在不触发报警器的前提下，
  • 最多可获得多少财宝

  • @author Liuhao
    解题思路：由于同时从相邻的两个房屋中盗取财宝就会触发报警器，故 a.若选择第i个房间盗取财宝，就一定不能选择第i-1个房间的盗取财宝
    b.若不选择第i个房间盗取财宝，则相当于值考虑前i-1个房间盗取财宝
    */
    public class StealPrecious {

    public static void main(String[] args) {

    Scanner scan=new Scanner(System.in);
System.out.println("请输入有多少个房间");
int n=scan.nextInt();
System.out.println("请输入每个房间的财宝");
int[] arr=new int[n];
for (int i = 0; i < arr.length; i++) {
    arr[i]=scan.nextInt();
}
int maxPrecious=getPrecious(arr);
System.out.println(maxPrecious);

    }
    public static int getPrecious(int arr[]) {

    if(arr==null) {
    return 0;
}
if(arr.length==1) {
    return arr[0];
}
List<Integer> dp=new ArrayList<>();
dp.add(arr[0]);
dp.add(Math.max(arr[0], arr[1]));
for(int i=2;i<arr.length;i++) {
    dp.add(Math.max(dp.get(i-1), dp.get(i-2)+arr[i]));
}
return dp.get(arr.length-1);

}

}
