    // TODO Auto-generated method stub

String getColumnLable(int n) {  

    String tmp = "";  
        // 按位找  
        int count = 1;  
        while (((int) (n / Math.pow(26, count-1)) > 0)) {  
            int w3 = ((int) (n % Math.pow(26, count)) / (int) Math.pow(26, count - 1));  
            char c = (char) (w3 + 65);  
            tmp = c + tmp;  
            count++;  
        }  
    return tmp;  
}

}
